from django.db import models
from producto.models import Producto  # Ajusta el path de importación según tu estructura de proyecto
from clientes.models import Cliente  # Ajusta el path de importación según tu estructura de proyecto

class Venta(models.Model):
    cliente = models.ForeignKey(Cliente, on_delete=models.CASCADE)
    fecha = models.DateTimeField(auto_now_add=True)

    def __str__(self):
        return f"Venta {self.id} - {self.fecha.strftime('%d/%m/%Y')}"

class DetalleVenta(models.Model):
    venta = models.ForeignKey(Venta, related_name='detalles', on_delete=models.CASCADE)
    producto = models.ForeignKey(Producto, on_delete=models.SET_NULL, null=True)
    cantidad = models.IntegerField()
    precio_unitario = models.DecimalField(max_digits=8, decimal_places=2)
    descuento = models.DecimalField(max_digits=5, decimal_places=2, default=0.00)

    @property
    def subtotal(self):
        return self.cantidad * self.precio_unitario

    @property
    def total(self):
        return self.subtotal - (self.subtotal * self.descuento / 100)

    def __str__(self):
        return f"Detalle de {self.venta}"
